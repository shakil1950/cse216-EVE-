/* Print array using recursive function*/ 
void arr(int a[],int s,int e)
{
	if(s>e)
	{
		return;
	}
	printf("%d\t",a[s]);
	arr(a,s+1,e);
}
void rearr(int b[],int start,int end)
{
	if(start>end)
	{
		return;
	}
	printf("%d\t",b[end]);
	rearr(b,start,end-1);
}

#include<stdio.h>

int main()
{
		int i,j,pos,n,temp,min;
	
	int a[100]={10,8,2,7,6,4,1,9};
	printf("\n ******Before sorting*****\n");
	
	for(i=0;i<8;i++)
	{
		printf("%d\t",a[i]);
	}
	for(i=0;i<8;i++)
	{
		min=a[i];
		pos=i;
		for(j=i+1;j<8;j++)
		{
			if(a[j]<min)
			{
				min=a[j];
				pos=j;
			
			}
		}
		temp=a[i];
		a[i]=a[pos];
		a[pos]=temp;
	}
	
	printf("\n ******After sorting******\n");
	for(i=0;i<8;i++)
	{
		printf("%d\t",a[i]);
	}
	return 0;
}
